# Zadanie 2
import random
sekretna_liczba = random.randint(1, 10)
i = 0
while i>=0:
    proba = int(input("Spróbuj zgadnąć sekretną liczbę w zakresie od 1 d0 10: "))
    i = i + 1
    if proba < 1 or proba > 10:
        print("Podaj liczbę z zakresu od 1 do 10.")
    else:
        if proba == sekretna_liczba:
            print(f"Gratulacje!! Odgadłeś sekretną liczbę! Miałeś {i} prób.")
            break
        else:
            print("Nie udało ci się odgadnąć.")
